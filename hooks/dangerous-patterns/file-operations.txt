# Directory removal

# Secure deletion utilities

srm[[:space:]]+

# Truncate files
truncate[[:space:]]+-s[[:space:]]*0

(^|[[:space:]])>[[:space:]]*/

# Move to /dev/null (data loss)
mv[[:space:]]+.*[[:space:]]+/dev/null

# Overwrite files
cp[[:space:]]+/dev/null[[:space:]]+
dd[[:space:]]+of=.*[[:space:]]+if=/dev/zero
dd[[:space:]]+of=.*[[:space:]]+if=/dev/urandom

# Archive with source removal
zip[[:space:]]+-m

tar[[:space:]]+--remove-files
7z[[:space:]]+d[[:space:]]

# Dangerous find -delete
find[[:space:]]+.*-delete
